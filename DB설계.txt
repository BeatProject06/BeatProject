create table userinfo(
    num int not null,
    id int(10) not null,
    nickname varchar(50) primary key,
    profile_img varchar(100),
    thumb_img varchar(100),
    register date
);

create table bbslist(
    no int not null,
    nickname varchar(50) not null,
    university varchar(50),
    dept varchar(50),
    bookname varchar(100) primary key,
    publisher varchar(50),
    author varchar(10),
    ISBN varchar(17),
    costprice int(6),
    saleprice int(6),
    foreign key(nickname) references user(nickname) on delete cascade
);
